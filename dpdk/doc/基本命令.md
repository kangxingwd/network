

# linux 的一些命令
```s
# 查看本机网络配置
ifconfig
# 查看当前网卡配置 
cat /etc/network/interfaces
# 查看物理cpu个数 
cat /proc/cpuinfo | grep "physical id" |sort | uniq |wc -l
# 查看每个物理cpu中的core（核数) 
cat /proc/cpuinfo | grep "cpu cores" | uniq
# 查看逻辑cpu的个数 
cat /proc/cpuinfo | grep "processor" | wc -l
# 查看网卡pci地址 
lspci | grep Ethernet
# 查看系统巨页 
cat /proc/meminfo | grep -i huge

# 设置系统时钟 
date -s
# 设置硬件时钟
hwclock --systohc

dmidecode memory

lspci -vvv

lscpu


```
